record.fields.Member[0] : null, memberName: record.fields['Member Name'] || '', startDate: record.fields['Start Date'] || '', endDate: record.fields['End Date'] || '', leaveType: record.fields['Leave Type'] || '', reason: record.fields.Reason || '', status: record.fields.Status || 'Pending', applicationDate: record.fields['Application Date'] || new Date().toISOString().split('T')[0] })); console.log(`Loaded from Airtable: ${members.length} members, ${attendance.length} attendance records`); showMessage('Data loaded successfully from Airtable!', 'success'); } catch (error) { console.error('Error loading from Airtable:', error); showMessage('Using local storage as backup', 'warning'); loadFromBrowserStorage(); } } // Save member to Airtable async function saveMemberToAirtable(member) { const fields = { 'Name': member.name, 'Phone': member.phone, 'Voice Part': member.voicePart, 'Email': member.email || '', 'Status': member.status || 'Active', 'Registered Date': member.registeredDate || new Date().toISOString().split('T')[0] }; const data = { fields: fields }; if (member.id) { // Update existing record return await airtableFetch(`${AIRTABLE_CONFIG.MEMBERS_TABLE}/${member.id}`, 'PATCH', data); } else { // Create new record return await airtableFetch(AIRTABLE_CONFIG.MEMBERS_TABLE, 'POST', data); } } // Save attendance to Airtable async function saveAttendanceToAirtable(record) { // First, get member name for display const member = members.find(m => m.id === record.memberId); const memberName = member ? member.name : ''; const fields = { 'Member': [record.memberId], 'Member Name': memberName, 'Date': record.date, 'Service Type': record.serviceType, 'Status': record.status || 'Present' }; const data = { fields: fields }; return await airtableFetch(AIRTABLE_CONFIG.ATTENDANCE_TABLE, 'POST', data); } // Save leave application to Airtable async function saveLeaveApplicationToAirtable(application) { const member = members.find(m => m.id === application.memberId); const memberName = member ? member.name : ''; const fields = { 'Member': [application.memberId], 'Member Name': memberName, 'Start Date': application.startDate, 'End Date': application.endDate, 'Leave Type': application.leaveType, 'Reason': application.reason, 'Status': application.status || 'Pending', 'Application Date': application.applicationDate || new Date().toISOString().split('T')[0] }; const data = { fields: fields }; return await airtableFetch(AIRTABLE_CONFIG.LEAVE_TABLE, 'POST', data); } // Enhanced registerMember function for Airtable async function registerMember() { try { const name = document.getElementById('memberName').value.trim(); const phone = document.getElementById('memberPhone').value.trim(); const voicePart = document.getElementById('voicePart').value; const email = document.getElementById('memberEmail').value.trim(); if (!name || !phone || !voicePart) { showMessage('Please fill in all required fields.', 'error'); return; } // Check if member already exists const existingMember = members.find(m => m.name.toLowerCase() === name.toLowerCase() || m.phone === phone ); if (existingMember) { showMessage('A member with this name or phone number already exists.', 'error'); return; } const newMember = { name: name, phone: phone, voicePart: voicePart, email: email, status: 'Active' }; // Save to Airtable const response = await saveMemberToAirtable(newMember); newMember.id = response.id; // Add to local array members.push(newMember); showMessage(`Registration successful!
